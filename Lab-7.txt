create or replace function fact(n in int)
return int
is 
result int:=1;
begin
for i in 1..n loop
    result:=result*i;
end loop;
return result;
end fact;

✅ 1. Check if a Number is Prime
CREATE OR REPLACE FUNCTION is_prime(n NUMBER)
RETURN VARCHAR2
IS
    i NUMBER := 2;
BEGIN
    IF n <= 1 THEN
        RETURN 'Not Prime';
    END IF;

    WHILE i <= SQRT(n) LOOP
        IF MOD(n, i) = 0 THEN
            RETURN 'Not Prime';
        END IF;
        i := i + 1;
    END LOOP;

    RETURN 'Prime';
END;
/

RETURN NUMBER
IS

✅ 4. Generate Fibonacci Sequence (Up to n terms)

(returns a string of numbers)

CREATE OR REPLACE FUNCTION fibonacci(n NUMBER)
RETURN VARCHAR2
IS
    a NUMBER := 0;
    b NUMBER := 1;
    temp NUMBER;
    result VARCHAR2(4000) := '';
BEGIN
    IF n <= 0 THEN
        RETURN 'Invalid Input';
    END IF;

    FOR i IN 1..n LOOP
        result := result || a || ' ';
        temp := a + b;
        a := b;
        b := temp;
    END LOOP;

    RETURN TRIM(result);
END;
/

✅ 5. Sum of Digits of a Number
CREATE OR REPLACE FUNCTION sum_of_digits(n NUMBER)
RETURN NUMBER
IS
    sumd NUMBER := 0;
    temp NUMBER := ABS(n);
BEGIN
    WHILE temp > 0 LOOP
        sumd := sumd + MOD(temp, 10);
        temp := FLOOR(temp / 10);
    END LOOP;

    RETURN sumd;
END;
/

✅ 6. Power of a Number (a^b)
CREATE OR REPLACE FUNCTION power_num(a NUMBER, b NUMBER)
RETURN NUMBER
IS
    result NUMBER := 1;
BEGIN
    FOR i IN 1..b LOOP
        result := result * a;
    END LOOP;

    RETURN result;
END;
/

✅ 7. Check if a Number is Even or Odd
CREATE OR REPLACE FUNCTION even_odd(n NUMBER)
RETURN VARCHAR2
IS
BEGIN
    IF MOD(n, 2) = 0 THEN
        RETURN 'Even';
    ELSE
        RETURN 'Odd';
    END IF;
END;
